#Исользовалось:
PHP 7.4.14, Node 15.10.0 (npm 7.5.3), MariaDB 10.4

#Установка и запуск:

Backend

    composer install
    php artisan serv

Frontend

    cd frontend
    npm install
    npm start

Database

    Дамп базы в файле - laravel-api.sql

#Настройка
настройка laravel, подключение к БД - /.env

настройка react (путь к api)- /frontend/.env
